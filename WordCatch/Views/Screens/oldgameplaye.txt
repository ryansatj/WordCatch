import SwiftUI
import Vision

struct Gameplay: View {
    var onExit: () -> Void = {}

    @State private var manager = HandDetectionModel()
    @State private var game = Game()

    @State private var cameraReady = false
    @State private var countdownValue: Int? = nil
    @State private var showHUD = false
    @State private var winnerScale: CGFloat = 0.6
    @State private var winnerOpacity: Double = 0

    private static let chains: [[VNHumanHandPoseObservation.JointName]] = [
        [.wrist, .thumbCMC, .thumbMP, .thumbIP, .thumbTip],
        [.wrist, .indexMCP, .indexPIP, .indexDIP, .indexTip],
        [.wrist, .middleMCP, .middlePIP, .middleDIP, .middleTip],
        [.wrist, .ringMCP, .ringPIP, .ringDIP, .ringTip],
        [.wrist, .littleMCP, .littlePIP, .littleDIP, .littleTip],
        [.indexMCP, .middleMCP, .ringMCP, .littleMCP],
    ]

    var body: some View {
        GeometryReader { geo in
            ZStack {
                CameraPreviewView(session: manager.session).ignoresSafeArea()

                handOverlay

                Rectangle()
                    .fill(.white.opacity(0.5))
                    .frame(width: 3)
                    .frame(maxHeight: .infinity)
                    .ignoresSafeArea()

                wordsLayer(in: geo.size)

                if showHUD { hudOverlay }

                if !cameraReady {
                    cameraLoadingOverlay
                        .transition(.opacity)
                }

                if let cv = countdownValue {
                    countdownOverlay(value: cv)
                        .transition(.opacity)
                }

                if let winner = game.winner {
                    winnerOverlay(winner: winner, size: geo.size)
                }
            } // LOCK LANDSCAPE WHEN PINDAH
            .onAppear {
                OrientationManager.shared.lockLandscape()
                manager.start()
                game.hands = { manager.tangan }
                game.size = geo.size
                startSequence()
            }
            .onChange(of: geo.size) { _, s in game.size = s }
            .onDisappear { // LOCK POOTRAIR PAS KELUAR
                manager.stop()
                game.stop()
                OrientationManager.shared.lockPortrait()
            }
        }
    }

    private var handOverlay: some View {
        Canvas { ctx, size in
            for hand in manager.tangan {
                let c: Color = hand.isOpen ? .green : .white.opacity(0.7)
                func pt(_ j: VNHumanHandPoseObservation.JointName) -> CGPoint? {
                    hand.points[j].map { CGPoint(x: $0.x * size.width, y: (1 - $0.y) * size.height) }
                }
                for chain in Self.chains {
                    var path = Path(); var on = false
                    for j in chain {
                        guard let p = pt(j) else { on = false; continue }
                        on ? path.addLine(to: p) : path.move(to: p); on = true
                    }
                    ctx.stroke(path, with: .color(c.opacity(0.85)), lineWidth: 2.5)
                }
                for (_, raw) in hand.points {
                    let p = CGPoint(x: raw.x * size.width, y: (1 - raw.y) * size.height)
                    ctx.fill(Path(ellipseIn: CGRect(x: p.x - 4, y: p.y - 4, width: 8, height: 8)),
                             with: .color(c))
                }
            }
        }
        .ignoresSafeArea()
        .allowsHitTesting(false)
    }

    private func wordsLayer(in size: CGSize) -> some View {
        ForEach(game.words) { w in
            let left = w.x < size.width / 2
            Text(w.text)
                .font(.system(size: 30, weight: .heavy, design: .rounded))
                .foregroundColor(.white)
                .shadow(color: .black.opacity(0.7), radius: 3, y: 2)
                .padding(.horizontal, 14).padding(.vertical, 8)
                .background(Capsule().fill(LinearGradient(
                    colors: left ? [.blue, .cyan] : [.red, .orange],
                    startPoint: .top, endPoint: .bottom)))
                .position(x: w.x, y: w.y)
        }
    }

    private var hudOverlay: some View {
        VStack {
            HStack(alignment: .top, spacing: 12) {
                ScorePill(label: "P1", score: game.ScoreP1, leading: true)

                Spacer()

                Text("First to \(game.winScore)")
                    .font(.system(size: 14, weight: .semibold, design: .rounded))
                    .foregroundColor(.white)
                    .padding(.horizontal, 12).padding(.vertical, 6)
                    .background(Capsule().fill(.black.opacity(0.45)))

                Spacer()

                ScorePill(label: "P2", score: game.ScoreP2, leading: false)
            }
            .padding(.horizontal, 24)
            .padding(.top, 16)

            Spacer()
        }
        .overlay(alignment: .topLeading) {
            Button(action: {
                manager.stop()
                game.stop()
                onExit()
            }) {
                Image(systemName: "xmark")
                    .font(.system(size: 16, weight: .bold))
                    .foregroundColor(.white)
                    .frame(width: 38, height: 38)
                    .background(Circle().fill(.black.opacity(0.5)))
            }
            .buttonStyle(.plain)
            .padding(.top, 60)
            .padding(.leading, 20)
        }
        .transition(.opacity)
    }

    private var cameraLoadingOverlay: some View {
        ZStack {
            Color.black.opacity(0.85).ignoresSafeArea()
            VStack(spacing: 18) {
                ProgressView()
                    .progressViewStyle(.circular)
                    .tint(.white)
                    .scaleEffect(1.4)
                Text("Preparing camera...")
                    .font(.system(size: 18, weight: .semibold, design: .rounded))
                    .foregroundColor(.white)
            }
        }
    }

    private func countdownOverlay(value: Int) -> some View {
        ZStack {
            Color.black.opacity(0.35).ignoresSafeArea()
            Text(value == 0 ? "GO!" : "\(value)")
                .font(.system(size: 140, weight: .black, design: .rounded))
                .foregroundColor(.white)
                .shadow(color: .black.opacity(0.5), radius: 8, y: 4)
                .id(value)
                .transition(.scale(scale: 1.6).combined(with: .opacity))
        }
        .animation(.spring(response: 0.45, dampingFraction: 0.65), value: value)
    }

    private func winnerOverlay(winner: Int, size: CGSize) -> some View {
        ZStack {
            Color.black.opacity(0.7).ignoresSafeArea()
            VStack(spacing: 24) {
                Text(winner == 0 ? "Player 1 Wins!" : "Player 2 Wins!")
                    .font(.system(size: 48, weight: .black, design: .rounded))
                    .foregroundColor(.white)
                    .multilineTextAlignment(.center)

                Text("🎉")
                    .font(.system(size: 64))

                HStack(spacing: 16) {
                    Button {
                        game.size = size
                        game.start()
                        runCountdown()
                    } label: {
                        Text("Play Again")
                            .font(.system(size: 20, weight: .bold, design: .rounded))
                            .foregroundColor(.black)
                            .padding(.horizontal, 28).padding(.vertical, 14)
                            .background(Capsule().fill(.white))
                    }
                    .buttonStyle(.plain)

                    Button {
                        manager.stop()
                        game.stop()
                        onExit()
                    } label: {
                        Text("Exit")
                            .font(.system(size: 20, weight: .bold, design: .rounded))
                            .foregroundColor(.white)
                            .padding(.horizontal, 28).padding(.vertical, 14)
                            .background(Capsule().stroke(.white, lineWidth: 2))
                    }
                    .buttonStyle(.plain)
                }
            }
            .scaleEffect(winnerScale)
            .opacity(winnerOpacity)
        }
        .onAppear {
            withAnimation(.spring(response: 0.5, dampingFraction: 0.6)) {
                winnerScale = 1.0
                winnerOpacity = 1.0
            }
        }
        .onDisappear {
            winnerScale = 0.6
            winnerOpacity = 0
        }
    }

    private func startSequence() {
        Task {
            try? await Task.sleep(for: .milliseconds(900))
            withAnimation(.easeOut(duration: 0.35)) { cameraReady = true }
            try? await Task.sleep(for: .milliseconds(150))
            runCountdown()
        }
    }

    private func runCountdown() {
        Task {
            for n in [3, 2, 1, 0] {
                withAnimation { countdownValue = n }
                try? await Task.sleep(for: .milliseconds(700))
            }
            withAnimation(.easeOut(duration: 0.2)) { countdownValue = nil }
            withAnimation(.easeOut(duration: 0.3)) { showHUD = true }
            game.start()
        }
    }
}

private struct ScorePill: View {
    let label: String
    let score: Int
    let leading: Bool

    var body: some View {
        HStack(spacing: 8) {
            Text(label)
                .font(.system(size: 13, weight: .bold, design: .rounded))
                .foregroundColor(.white.opacity(0.7))
                .padding(.horizontal, 8).padding(.vertical, 3)
                .background(Capsule().fill(leading ? Color.blue.opacity(0.6) : Color.red.opacity(0.6)))

            Text("\(score)")
                .font(.system(size: 22, weight: .heavy, design: .rounded))
                .foregroundColor(.white)
                .contentTransition(.numericText())
                .animation(.spring(response: 0.3, dampingFraction: 0.6), value: score)
        }
        .padding(.horizontal, 14).padding(.vertical, 8)
        .background(Capsule().fill(.black.opacity(0.5)))
    }
}
